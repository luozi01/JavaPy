func quickSort(int[] list){
    less := int[]
    equal := int[]
    greater := int[]

    if list.length > 1{
        pivot := list[0]
        for num in list {
            if num < pivot{
                less.add(num)
            }
            if num == pivot {
                equal.add(num)
            }
            if num > pivot {
                greater.add(num)
            }
        }
        return (quickSort(less) + equal + quickSort(greater))
    }
    else {
        return list
    }
}
